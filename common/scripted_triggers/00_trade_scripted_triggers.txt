already_trade_partners = {
    custom_tooltip = {
        text = "Already trading in this category"
        OR = {
            AND = {
                $country_B$ = {
                    is_target_in_variable_list = {
                        name = list_of_trade_partners_$category$
                        target = $country_A$
                    }
                }
                $country_A$ = {
                    is_target_in_variable_list = {
                        name = list_of_trade_partners_$category$
                        target = $country_B$
                    }
                }
            }
        }
    }
}

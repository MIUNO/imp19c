	}
    is_shown = { # Can be omitted as always true
    	always = yes
    }

		# Triggered in AI script, not through the GUI
		always = no
	}

    	}
    }
    effect = {

    	DIP_TRADE_toggle_trade_agreement = {
    		category = all_categories

    	}

    }

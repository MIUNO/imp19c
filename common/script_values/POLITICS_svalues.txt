# "National spirit" svalues, representing the values of a country according to its government
# All scopes are Country

SPIRIT_piety = {
	# The religious piety of a country
	value = 0
	# FROM STATE RELIGION LAW
	if = {
		limit = {
			has_law = enforced_state_religion_law
		}
		add = 10
	}
	else_if = {
		limit = {
			has_law = nominal_state_religion_law
		}
		add = 3
	}
	else_if = {
		limit = {
			has_law = state_atheism
		}
		subtract = 10
	}

	# FROM UNIVERSITY LAW
	if = {
		limit = {
			has_law = law_university_religious_colleges
		}
		add = 5
	}
	else_if = {
		limit = {
			OR = {
				has_law = law_university_secular_instutions
				has_law = law_university_technical_colleges
			}
		}
		subtract = 2
	}

}

SPIRIT_nationalism = {
	# The degree to which a country will seek to protect and influence pops of its culture
	value = 0
	# FROM CULTURAL PROTECTIONS LAW
	if = {
		limit = {
			has_law = dominant_culture_primacy
		}
		add = 10
	}
	else_if = {
		limit = {
			has_law = limited_cultural_recognition
		}
		add = 5
	}
	else_if = {
		limit = {
			has_law = cultural_federalism
		}
		add = 2
	}
	else_if = {
		limit = {
			has_law = multiculturalism_law
		}
		subtract = 5
	}

	# FROM REGIONAL GOVERNMENTS LAW
	if = {
		limit = {
			has_law = feudal_regional_governments
		}
		subtract = 7
	}
	else_if = {
		limit = {
			has_law = devolved_regional_governments
		}
		subtract = 5
	}

	if = {
		limit = {
			trigger_if = {
				limit = {
					is_subject = yes
				}
				OR = {
					is_subject_type = client_colony
					is_subject_type = autonomous_colony
					is_subject_type = territory
				}
			}
		}
		value = 0
	}

}

SPIRIT_traditionalism = {
	# The degree to whcih a country seeks to protect traditional structures of power
	value = 0
	#### From government type
	if = {
		limit = {
			OR = {
				government = absolute_kingdom
				government = absolute_duchy
				government = absolute_grand_duchy
				government = absolute_principality
				government = absolute_county
				government = imperial_monarchy
				government = autocratic_regency
				government = elective_theocracy
				government = theocratic_protectorate
				government = catholic_papacy
				government = hereditary_theocracy
				government = militant_theocracy
				government = fascist_dictatorship
				government = coregency
			}
		}
		add = 10
	}
	else_if = {
		limit = {
			OR = {
				government = hereditary_dictatorship
				government = dikastocracy
				government = oligarchy
				government = viceroyalty
				government = megacorporation
			}
		}
		add = 8
	}
	else_if = {
		limit = {
			OR = {
				government = personalist_dictatorship
				government = stratocracy
			}
			add = 5
		}
	}
	# Negative
	else_if = {
		limit = {
			OR = {
				government = constitutional_parliament
				government = constitutional_republic
			}
			subtract = 2
		}
	}
	else_if = {
		limit = {
			OR = {
				government = revolutionary_republic
				government = soviet_republic
			}
			subtract = 10
		}
	}

	#### From policies

	# Religion
	if = {
		limit = {
			has_law = enforced_state_religion_law
		}
		add = 10
	}
	else_if = {
		limit = {
			has_law = nominal_state_religion_law
		}
		add = 3
	}
	else_if = {
		limit = {
			has_law = state_atheism
		}
		subtract = 10
	}

	# Civil rights
	if = {
		limit = {
			has_law = wealth_based_rights
		}
		add = 3
	}
	else_if = {
		limit = {
			has_law = constitutional_rights
		}
		subtract = 3
	}

	# Constitutional monarchy
	if = {
		limit = {
			has_law = administrative_monarchy
		}
		add = 8
	}
	else_if = {
		limit = {
			has_law = advisory_monarchy
		}
		add = 4
	}
	else_if = {
		limit = {
			has_law = no_monarchy
		}
		subtract = 4
	}

	# Women's rights
	if = {
		limit = {
			has_law = women_second_class_status
		}
		add = 2
	}
	else_if = {
		limit = {
			has_law = womens_suffrage_law
		}
		subtract = 2
	}

	# University law
	if = {
		limit = {
			has_law = law_university_religious_colleges
		}
		add = 2
	}
}

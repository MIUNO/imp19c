namespace = scheme
# Each scheme requires the scope scheme_target to be set in the immediate, prior to adding the scheme.
# Scheme title and description are associated with the ID, and stored in custom localization
#

# TODO: Mod-specific schemes

scheme.1 = {
	type = minor_character_event
	title = "scheme.1.t"
	desc = "scheme.1.desc"
	
	picture = dagger_behind_back
	
	left_portrait = scope:thief
	
	interface_lock = no
	
	trigger = {
		is_alive = yes
	}
	
	option = {
		name = "scheme.1.a"
		trigger = {
			NOT = {
				has_trait = lunatic
			}
		}
		add_gold = -200
		scope:thief = {
			add_gold = 200
		}
	}
	
	option = {
		name = "scheme.1.b"
		custom_tooltip = "high_martial_character_tt"
		trigger = {
			martial >= 10
			NOT = {
				has_trait = lunatic
			}
		}
		highlight = yes
		add_gold = -200
		scope:thief = {
			add_gold = 200
			if = {
				limit = {
					health >= 10
				}
				add_health = -10
			}
			else = {
				death = {
					death_reason = death_while_stealing
				}
			}
		}
	}
	
	option = {
		name = "scheme.1.c"
		custom_tooltip = "lunatic_bad_tooltip"
		trigger = {
			has_trait = lunatic
		}
		exclusive = yes
		highlight = yes
		add_gold = -400
		scope:thief = {
			add_gold = 400
		}
	}
	
}
